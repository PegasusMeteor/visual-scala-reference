    \node (a1) {$a_1$}; &
    \node (a2) {$a_2$}; &
    \elementsbetween &
    \node (ai) {$a_i$}; &
    \node (ai+1) {$a_{i+1}$}; &
    \elementsafter \\

    \node (b2) {$a_2$}; &
    \elementsbetween &
    \node (bi) {$a_i$}; \\
};
