    \node (a1) {$a_1$}; &
    \node (a2) {$a_2$}; &
    \node [elements between]; &
    \node (ai) {$a_i$}; &
    \node [elements after]; &

\foreach \i/\r in {1/\neq,2/\neq,i/=}{
    \draw [subflow ->] (a\i.north) -- +(0, .5)
        node [subelement, above] {$\texttt{a}\r a_\i$};
}

    node [element, below] {$i$};
