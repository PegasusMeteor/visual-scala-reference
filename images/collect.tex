    \node (ai) {$a_i$}; &
    \node (ai+1) {$a_{i+1}$}; &
    \elementsbetween &
    \node (an-1) {$a_{n-1}$}; &
    \node (an) {$a_n$}; \\
};

\matrix [collection, below=3 of A] {
    \node (b2) {$b_1$}; &
