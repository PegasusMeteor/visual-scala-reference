    \node (a1) {$a_1$}; &
    \node (a2) {$a_2$}; &

    \elementsafter &

    \node (f\i) [subfunction, below=1 of a\i] {\texttt{f}};

}

\node (b) [some={$b$}, below=1 of fi];
