\matrix (A) [collection] {
    \node [elements before]; &
    \node (an-i) {$a_{n - i}$}; &
    \node [elements between]; &
    \node (an-1) {$a_{n - 1}$}; &
    \node (an) {$a_n$}; &
\\ };

\foreach \i/\r in {n/\neq,n-1/\neq,n-i/=}{
    \draw [subflow ->] (a\i.north) -- +(0, .5)
        node (c) [subelement, above] {$\texttt{a}\r a_{\i}$};
}

\draw [<- flow] (an.north) arc [start angle=160, delta angle=-60, x radius=.75, y radius=.5];

\draw [flow ->, dashed] (an-1.north) arc [start angle=15, delta angle=60, x radius=.75, y radius=.5];
\draw [<- flow, dashed] (an-i.north) arc [start angle=160, delta angle=-60, x radius=.75, y radius=.5];

    node (b) [element, below] {$n - i$};
