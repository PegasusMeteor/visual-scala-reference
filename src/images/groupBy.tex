  \elem{a}{1} &
  \elem{a}{2} &

  \elem{a}{i+1} &
  \helem{between} &
  \elem{a}{n} \\
};

\node [big arrow, below=\cellheight - .5\bigarrowwidth of A, anchor=west, rotate=-90];

\foreach \i/\k in {
  1/k_1,
  2/k_2,
  3/k_1,
  i/k_m,
  i+1/k_2,

    node (f\i) [predicate evaluation, above=.5 of a\i] {$\param{f}(a_{\i})$}

    node (fk\i) [predicate evaluation, above=.5 of f\i] {$\k$}

}
